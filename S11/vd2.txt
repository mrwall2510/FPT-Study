#include <stdio.h>

void main(){
    int ary[10];
    int i, total, max, min;
    for(i=0;i<10;i++){
        printf("\nEnter valua: %d: ", i+1);
        scanf("%d", &ary[i]);
    }
    max = ary[0];
    for(i=1;i<10;i++){
        if (ary[i]>max)
            max=ary[i];
    }
    printf("\nMax value entered was %d", max);
    min = ary[0];
    for(i=1;i<10;i++){
        if (ary[i]<min)
            min=ary[i];
    }
    printf("\nMin value entered was %d", min);
    for (i=0, total=0;i<10;i++)
        total=total+ary[i];
    printf("\nThe average of the element of ary is %d",total/i);
}
