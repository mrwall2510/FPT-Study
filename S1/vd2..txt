#include <stdio.h>

int main() {
    int a, b, c, sum, Avg;
    printf("Diem Vat Ly: ");
    scanf("%d", &a);
    printf("Diem Hoa Hoc: ");
    scanf("%d", &b);
    printf("Diem Sinh Hoc: ");
    scanf("%d", &c);
    sum = a + b + c;
    Avg = sum / 3;
    printf("Diem Tong= %d\n", sum);
    printf("Diem Trung Binh= %d", Avg);
    return 0;
}
